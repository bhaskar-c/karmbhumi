  </div>
  
  <div class="hindi">

  </div>
  
  <div class="hindi">

</div>

<div class="doha">

  </div>
  
  <div class="hindi">

  </div>
</div>

<div class="doha">

  </div>
  
  <div class="hindi">

</div>

<div class="doha">

  </div>
  
  <div class="hindi">

  </div>
</div>

<div class="doha">

  <div class="hindi">
